/**
 * Require admin role. Use AFTER authenticate.
 * Usage: router.get('/admin/foo', authenticate, requireAdmin, handler);
 */
function requireAdmin(req, res, next) {
  // Pool reference comes from the route file
  if (!req.customerId) {
    return res.status(401).json({ error: 'Authentication required' });
  }
  // The actual admin check happens in the route handler with the pool
  next();
}

module.exports = { authenticate, generateToken, requireAdmin };

/**
 * Helper to verify admin status from database
 * Returns the customer record if admin, throws otherwise
 */
async function verifyAdmin(pool, customerId) {
  const result = await pool.query(
    'SELECT id, email, is_admin, role, suspended FROM customers WHERE id = $1',
    [customerId]
  );
  
  if (result.rows.length === 0) {
    throw new Error('User not found');
  }
  
  const user = result.rows[0];
  if (user.suspended) {
    throw new Error('Account suspended');
  }
  if (!user.is_admin) {
    throw new Error('Admin access required');
  }
  
  return user;
}

module.exports = { authenticate, generateToken, requireAdmin, verifyAdmin };

class AuditLog {
  constructor(pool) {
    this.pool = pool;
  }

  /**
   * Log an admin action
   */
  async log(adminId, adminEmail, action, targetType, targetId, details = {}, req = null) {
    try {
      const ip = req?.ip || req?.headers?.['x-forwarded-for']?.split(',')[0] || null;
      const userAgent = req?.headers?.['user-agent'] || null;

      await this.pool.query(
        `INSERT INTO admin_audit_log 
         (admin_id, admin_email, action, target_type, target_id, details, ip_address, user_agent)
         VALUES ($1, $2, $3, $4, $5, $6, $7, $8)`,
        [adminId, adminEmail, action, targetType, targetId, JSON.stringify(details), ip, userAgent]
      );
    } catch (err) {
      console.error('Audit log error (non-fatal):', err.message);
      // Don't throw — audit failures shouldn't break the action
    }
  }

  async getRecent(limit = 50) {
    const result = await this.pool.query(
      `SELECT * FROM admin_audit_log ORDER BY created_at DESC LIMIT $1`,
      [limit]
    );
    return result.rows;
  }

  async getForTarget(targetType, targetId) {
    const result = await this.pool.query(
      `SELECT * FROM admin_audit_log 
       WHERE target_type = $1 AND target_id = $2 
       ORDER BY created_at DESC LIMIT 100`,
      [targetType, targetId]
    );
    return result.rows;
  }
}

module.exports = AuditLog;

      let orderBy = 'created_at DESC';
      if (sort === 'best') orderBy = 'performance_score DESC NULLS LAST';
      if (sort === 'engagement') orderBy = `((engagement->>'likes')::int + (engagement->>'comments')::int + (engagement->>'shares')::int) DESC`;

      let query = `
        SELECT id, content_type, caption, media_url, status, scheduled_date, posted_at,
               engagement, engagement_by_platform, performance_score, platforms,
               image_provider, created_at
        FROM posts 
        WHERE customer_id = $1 AND status = 'posted'
      `;
      const params = [req.customerId];
      let p = 1;

      if (platform) {
        p++;
        query += ` AND $${p} = ANY(SELECT jsonb_array_elements_text(platforms))`;
        params.push(platform);
      }
      if (contentType) {
        p++;
        query += ` AND content_type = $${p}`;
        params.push(contentType);
      }

      query += ` ORDER BY ${orderBy} LIMIT $${p + 1}`;

        shares: { value: eng.shares || 0, avg: Math.round(avg.avg_shares), diff: ((eng.shares || 0) - avg.avg_shares) / Math.max(1, avg.avg_shares) * 100 },
      };

      const hashtagCount = (post.hashtags || []).length;
      const postHour = post.posted_at ? new Date(post.posted_at).getHours() : null;

        });

      if (newPassword.length < 8) return res.status(400).json({ error: 'Password must be 8+ chars' });

import Layout from '../../components/Layout';
import { Card, Button, Badge, EmptyState } from '../../components/ui';
import { useTheme } from '../../lib/theme';
import { adminAPI } from '../../lib/api';
import { Search, ChevronRight, Users, Filter } from 'lucide-react';

export default function AdminCustomers() {
  const router = useRouter();
  const { t } = useTheme();
  const [mounted, setMounted] = useState(false);
  const [data, setData] = useState({ customers: [], total: 0 });
  const [search, setSearch] = useState('');
  const [filterPlan, setFilterPlan] = useState('');
  const [filterSuspended, setFilterSuspended] = useState('');
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    setMounted(true);
    if (!localStorage.getItem('token')) {
      router.replace('/login');
      return;
    }
    load();
  }, []);

  useEffect(() => {
    if (mounted) {
      const t = setTimeout(load, 300);
      return () => clearTimeout(t);
    }
  }, [search, filterPlan, filterSuspended]);

  const load = async () => {
    setLoading(true);
    try {
      const params = { limit: 100 };
      if (search) params.search = search;
      if (filterPlan) params.plan = filterPlan;
      if (filterSuspended) params.suspended = filterSuspended;
      const res = await adminAPI.listCustomers(params);
      setData(res.data);
    } catch (err) {
      if (err.response?.status === 403) router.replace('/dashboard');
    } finally {
      setLoading(false);
    }
  };

  if (!mounted) return null;

  return (
    <Layout title="Customers" subtitle={`${data.total} total`}>
      {/* Filters */}
      <div style={{ display: 'flex', gap: 12, marginBottom: 20, flexWrap: 'wrap' }}>

            placeholder="Search by email or business name..."

            style={{
              width: '100%',
              padding: '10px 12px 10px 36px',
              background: t.input,
              border: `1px solid ${t.borderStrong}`,
              borderRadius: 8,
              color: t.text,
              fontSize: 13,
            }}
          />
        </div>

        <select
          value={filterPlan}
          onChange={(e) => setFilterPlan(e.target.value)}
          style={{
            padding: '10px 12px',
            background: t.input,
            border: `1px solid ${t.borderStrong}`,
            borderRadius: 8,
            color: t.text,
            fontSize: 13,
            cursor: 'pointer',
          }}
        >
          <option value="">All plans</option>
          <option value="trial">Trial</option>
          <option value="starter">Starter</option>
          <option value="professional">Professional</option>
          <option value="premium">Premium</option>
        </select>

        <select
          value={filterSuspended}
          onChange={(e) => setFilterSuspended(e.target.value)}
          style={{
            padding: '10px 12px',
            background: t.input,
            border: `1px solid ${t.borderStrong}`,
            borderRadius: 8,
            color: t.text,
            fontSize: 13,
            cursor: 'pointer',
          }}
        >
          <option value="">All</option>
          <option value="false">Active only</option>
          <option value="true">Suspended only</option>
        </select>
      </div>

      <Card padding={0}>
        {loading ? (
          <div style={{ padding: 60, textAlign: 'center', color: t.textMuted }}>Loading...</div>
        ) : data.customers.length === 0 ? (
          <EmptyState icon={Users} title="No customers found" />

                      padding: '12px 20px',

                      letterSpacing: '0.05em',

                  >
                    {h}
                  </th>
                ))}

                <tr
                  key={c.id}
                  onClick={() => router.push(`/admin/customers/${c.id}`)}

function CreditsModal({ onClose, onSubmit, t }) {
  const [amount, setAmount] = useState('');
  const [reason, setReason] = useState('');

  const submit = () => {
    const n = parseInt(amount);
    if (!n || isNaN(n)) return alert('Enter a valid number (use negative to deduct)');
    if (!reason || reason.length < 3) return alert('Reason required');
    onSubmit(n, reason);
  };

  return (
    <div onClick={onClose} style={{ position: 'fixed', inset: 0, background: 'rgba(0,0,0,0.7)', zIndex: 100, display: 'flex', alignItems: 'center', justifyContent: 'center', padding: 20 }}>
      <div onClick={(e) => e.stopPropagation()} style={{ background: t.card, borderRadius: 12, padding: 24, maxWidth: 440, width: '100%' }}>
        <h3 style={{ fontSize: 18, fontWeight: 700, color: t.text, marginBottom: 16 }}>Adjust credits</h3>
        <p style={{ fontSize: 13, color: t.textMuted, marginBottom: 16 }}>
          Use a negative number to deduct credits. Positive to grant.
        </p>
        <input
          type="number"
          placeholder="e.g. 50 or -10"
          value={amount}
          onChange={(e) => setAmount(e.target.value)}
          style={{ width: '100%', padding: '10px 12px', background: t.input, border: `1px solid ${t.borderStrong}`, borderRadius: 8, color: t.text, marginBottom: 12 }}
        />
        <textarea
          placeholder="Reason (visible in audit log)"
          value={reason}
          onChange={(e) => setReason(e.target.value)}
          rows={3}
          style={{ width: '100%', padding: '10px 12px', background: t.input, border: `1px solid ${t.borderStrong}`, borderRadius: 8, color: t.text, marginBottom: 16, fontFamily: 'inherit', resize: 'vertical' }}
        />
        <div style={{ display: 'flex', gap: 8, justifyContent: 'flex-end' }}>
          <button onClick={onClose} style={{ padding: '8px 16px', background: 'transparent', border: `1px solid ${t.border}`, borderRadius: 8, color: t.text, fontSize: 13, fontWeight: 600, cursor: 'pointer' }}>Cancel</button>
          <button onClick={submit} style={{ padding: '8px 16px', background: t.primary, border: 'none', borderRadius: 8, color: '#fff', fontSize: 13, fontWeight: 600, cursor: 'pointer' }}>Apply</button>
        </div>
      </div>
    </div>
  );
}

function SuspendModal({ onClose, onSubmit, t }) {
  const [reason, setReason] = useState('');

  return (
    <div onClick={onClose} style={{ position: 'fixed', inset: 0, background: 'rgba(0,0,0,0.7)', zIndex: 100, display: 'flex', alignItems: 'center', justifyContent: 'center', padding: 20 }}>
      <div onClick={(e) => e.stopPropagation()} style={{ background: t.card, borderRadius: 12, padding: 24, maxWidth: 440, width: '100%' }}>
        <h3 style={{ fontSize: 18, fontWeight: 700, color: t.text, marginBottom: 16 }}>Suspend account</h3>
        <p style={{ fontSize: 13, color: t.textMuted, marginBottom: 16 }}>
          The customer will be blocked from logging in until reactivated.
        </p>
        <textarea
          placeholder="Reason for suspension..."
          value={reason}
          onChange={(e) => setReason(e.target.value)}
          rows={3}
          style={{ width: '100%', padding: '10px 12px', background: t.input, border: `1px solid ${t.borderStrong}`, borderRadius: 8, color: t.text, marginBottom: 16, fontFamily: 'inherit', resize: 'vertical' }}
        />
        <div style={{ display: 'flex', gap: 8, justifyContent: 'flex-end' }}>

          <button onClick={() => reason && onSubmit(reason)} style={{ padding: '8px 16px', background: t.error, border: 'none', borderRadius: 8, color: '#fff', fontSize: 13, fontWeight: 600, cursor: 'pointer' }}>Suspend</button>
        </div>
      </div>
    </div>
  );
}

function ComparisonCard({ t, icon: Icon, label, value, avg, diff }) {
  const isUp = diff > 0;
  const isDown = diff < 0;
  return (
    <Card>
      <div style={{ display: 'flex', alignItems: 'center', gap: 8, color: t.textMuted, fontSize: 12, fontWeight: 500 }}>
        <Icon size={14} />
        {label}
      </div>
      <div style={{ fontSize: 32, fontWeight: 800, color: t.text, marginTop: 8, fontFamily: 'monospace' }}>{value}</div>
      <div style={{ display: 'flex', alignItems: 'center', gap: 6, marginTop: 6 }}>
        {isUp && <TrendingUp size={12} style={{ color: t.success }} />}
        {isDown && <TrendingDown size={12} style={{ color: t.error }} />}
        <span style={{ fontSize: 11, color: isUp ? t.success : isDown ? t.error : t.textMuted, fontWeight: 600 }}>
          {diff > 0 ? '+' : ''}{Math.round(diff)}% vs avg ({avg})
        </span>
      </div>
    </Card>
  );
}

      </div>
      <div style={{ fontSize: 18, fontWeight: 700, color: t.text, fontFamily: 'monospace', marginTop: 2 }}>{value}</div>
    </div>
  );
}

## What this completes

This prompt closes the loop on production readiness. Three things stand out:

**The admin portal works because of the audit log.** Every credit adjustment, suspension, and role change writes to `admin_audit_log` with the admin's email, the action, and the IP — so when something goes wrong (or a customer disputes a charge), you can trace exactly what happened. That's the difference between a toy admin panel and a production-grade one.

**Post performance uses real comparison.** Instead of just showing raw numbers, the detail page contextualizes each post against the account's own average — so a customer with 50 likes per post sees their 80-like post as "+60% vs avg" and feels the win, while a customer averaging 200 sees their 80-like post as a flag to investigate. The `post_engagement_snapshots` table is ready to receive real data when you wire up the platform APIs (Facebook Graph, Instagram, GBP).

**The hidden production gaps are addressed.** Suspended users were previously a security hole (the flag existed but nothing checked it on login). Email reset existed only as a stub. Rate limits were lenient enough to allow brute force. All three are now closed in this round.
